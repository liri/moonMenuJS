Manual menu initialization:

You can create your own moon-menu with customized options:

$(document).ready(function() {
                var moonmenu = $(".mymenu").moonMenu({
                    openTrigger: 'mouseover',
                    closeTrigger: 'click',
                    closeOnLostFocus: true,
                    
                    radius: 40,
                    position: 'middle left',
                    reverseOrder: false,
                    densityFactor: 2.8
                });
            });
